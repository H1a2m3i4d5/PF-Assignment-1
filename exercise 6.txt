def display_divisible_by_5(numbers):
    divisible_by_5 = [num for num in numbers if num % 5 == 0]
    return divisible_by_5

# Example usage:
numbers_list = [10, 25, 30, 45, 50, 55, 60]
divisible_by_5_list = display_divisible_by_5(numbers_list)
print("Numbers divisible by 5:", divisible_by_5_list)
